    <script>
        let growthChart = null;
        let currentMode = '7'; // 預設顯示 7% 金雞母

        // 全局財務數據存儲
        let currentCalcData = {
            age: 30,
            retire: 60,
            saved: 0,
            finalFund: 14113600,
            saveFlat: 39204,
            save7: 12011,
            save10: 6242
        };

        document.addEventListener('DOMContentLoaded', function() {
            const inputs = {
                age: { num: document.getElementById('num-age'), range: document.getElementById('range-age'), txt: document.getElementById('val-age') },
                retire: { num: document.getElementById('num-retire'), range: document.getElementById('range-retire'), txt: document.getElementById('val-retire') },
                expense: { num: document.getElementById('num-expense'), txt: document.getElementById('val-expense') },
                saved: { num: document.getElementById('num-saved'), txt: document.getElementById('val-saved') }
            };

            function setupSync(elementObj) {
                if(!elementObj.range) return;
                elementObj.num.addEventListener('input', function() { elementObj.range.value = this.value; calculateRetirement(); });
                elementObj.range.addEventListener('input', function() { elementObj.num.value = this.value; calculateRetirement(); });
            }

            setupSync(inputs.age);
            setupSync(inputs.retire);
            inputs.expense.num.addEventListener('input', calculateRetirement);
            inputs.saved.num.addEventListener('input', calculateRetirement);

            calculateRetirement();
        });

        function calculateRetirement() {
            const age = parseInt(document.getElementById('num-age').value) || 0;
            const retire = parseInt(document.getElementById('num-retire').value) || 0;
            const expense = parseInt(document.getElementById('num-expense').value) || 0;
            const saved = parseInt(document.getElementById('num-saved').value) || 0;
            
            const swr = 6.0; 
            const inflationRate = 0.02; 
            const govSubsidy = 20000; 

            document.getElementById('val-age').innerText = age;
            document.getElementById('val-retire').innerText = retire;
            document.getElementById('val-expense').innerText = expense.toLocaleString();
            document.getElementById('val-saved').innerText = saved.toLocaleString();

            const resultArea = document.getElementById('result-area');

            if (retire > age && expense > 0) {
                const yearsToRetire = retire - age;
                const futureExpense = Math.round(expense * Math.pow(1 + inflationRate, yearsToRetire));
                const gapExpense = Math.max(0, futureExpense - govSubsidy);
                const rawTotalFund = Math.round((gapExpense * 12) / (swr / 100));
                const finalFund = Math.max(0, rawTotalFund - saved);
                const totalMonths = yearsToRetire * 12;
                
                const saveFlat = Math.round(finalFund / totalMonths);

                function getPMT(rate, months, target) {
                    if(target <= 0) return 0;
                    const monthlyRate = rate / 12;
                    const factor = Math.pow(1 + monthlyRate, months) - 1;
                    return Math.round(target * (monthlyRate / factor));
                }

                const saveCompound7 = getPMT(0.07, totalMonths, finalFund);
                const saveCompound10 = getPMT(0.10, totalMonths, finalFund);

                // 更新前端文字數據
                document.getElementById('res-years-to-retire').innerText = yearsToRetire;
                document.getElementById('res-future-expense').innerText = futureExpense.toLocaleString();
                document.getElementById('res-gap-expense').innerText = gapExpense.toLocaleString();
                document.getElementById('res-raw-total-fund').innerText = rawTotalFund.toLocaleString();
                document.getElementById('res-already-saved').innerText = saved.toLocaleString();
                document.getElementById('res-final-fund').innerText = finalFund.toLocaleString();
                
                document.getElementById('res-save-flat').innerText = saveFlat.toLocaleString() + " 元/月";
                document.getElementById('res-save-compound-7').innerText = saveCompound7.toLocaleString() + " 元/月";
                document.getElementById('res-save-compound-10').innerText = saveCompound10.toLocaleString() + " 元/月";

                // 儲存全域變數供圖表重繪使用
                currentCalcData = { age, retire, saved, finalFund, saveFlat, save7: saveCompound7, save10: saveCompound10 };

                // 完美執行指定的動態碎碎念文案（中性通用文字）
                document.getElementById('coach-comment').innerHTML = `距離退休還有 <strong>${yearsToRetire}</strong> 年，時間是我們最大的優勢。純儲蓄每個月要存 <strong>${saveFlat.toLocaleString()}</strong> 元，而啟動 10% 投資理財加速器後只需要 <strong>${saveCompound10.toLocaleString()}</strong> 元。少辛苦那幾十萬分的力氣，留給陪伴家人與享受當下的生活吧！`;

                resultArea.classList.add('active');
                
                // 更新或創建圖表
                updateChart();
            } else {
                resultArea.classList.remove('active');
            }
        }

        // 切換圖表模式 A(flat) / B(7) / C(10)
        function switchChartMode(mode) {
            currentMode = mode;
            
            // 移除所有 active 樣式
            document.getElementById('box-flat').classList.remove('active');
            document.getElementById('box-7').classList.remove('active');
            document.getElementById('box-10').classList.remove('active');

            // 根據模式啟用樣式與標題文字
            const titleTxt = document.getElementById('chart-current-mode');
            if(mode === 'flat') {
                document.getElementById('box-flat').classList.add('active');
                titleTxt.innerText = "目前顯示：0% 純儲蓄規劃曲線";
            } else if(mode === '7') {
                document.getElementById('box-7').classList.add('active');
                titleTxt.innerText = "目前顯示：7% 穩健理財金雞母曲線";
            } else if(mode === '10') {
                document.getElementById('box-10').classList.add('active');
                titleTxt.innerText = "目前顯示：10% 強效投資理財加速器曲線";
            }

            updateChart();
        }

        // 動態計算複利成長數據並繪製圖表
        function updateChart() {
            const ctx = document.getElementById('growthChart').getContext('2d');
            
            const startAge = currentCalcData.age;
            const endAge = currentCalcData.retire;
            const startMoney = currentCalcData.saved; // 起點從目前已備資金開始
            const finalFundGap = currentCalcData.finalFund;

            let labels = [];
            let chartData = [];
            
            let rate = 0;
            let monthlyInvest = 0;

            if(currentMode === 'flat') {
                rate = 0;
                monthlyInvest = currentCalcData.saveFlat;
            } else if(currentMode === '7') {
                rate = 0.07;
                monthlyInvest = currentCalcData.save7;
            } else if(currentMode === '10') {
                rate = 0.10;
                monthlyInvest = currentCalcData.save10;
            }

            let currentBalance = startMoney;
            labels.push(startAge + " 歲");
            chartData.push(Math.round(currentBalance / 10000)); // 以「萬元」為單位

            const monthlyRate = rate / 12;

            // 逐年推算資產包
            for(let age = startAge + 1; age <= endAge; age++) {
                labels.push(age + " 歲");
                // 每個月滾動複利 12 次
                for(let m = 0; m < 12; m++) {
                    currentBalance = (currentBalance + monthlyInvest) * (1 + monthlyRate);
                }
                chartData.push(Math.round(currentBalance / 10000));
            }

            // 如果圖表已存在，先銷毀後再重建（Chart.js 最佳實踐）
            if (growthChart) {
                growthChart.destroy();
            }

            // 設定圖表線條顏色
            let lineColor = '#d4867a'; // 7% 玫瑰金
            if(currentMode === 'flat') lineColor = '#564743'; // 定存深棕
            if(currentMode === '10') lineColor = '#2b5c8f'; // 10% 科技藍

            growthChart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: labels,
                    datasets: [{
                        label: '預估累積資產 (萬元)',
                        data: chartData,
                        borderColor: lineColor,
                        backgroundColor: lineColor + '1A', // 透明背景填滿
                        borderWidth: 3,
                        pointBackgroundColor: lineColor,
                        pointHoverRadius: 6,
                        fill: true,
                        tension: 0.3 // 讓曲線變溫柔圓滑
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: { display: false },
                        tooltip: {
                            callbacks: {
                                label: function(context) {
                                    let ageStr = context.label;
                                    let val = context.raw;
                                    let targetTotal = Math.round((startMoney + finalFundGap) / 10000);
                                    let pct = targetTotal > 0 ? Math.round((val / targetTotal) * 100) : 100;
                                    return `累積資產: ${val} 萬元 (達標率 ${pct}%)`;
                                }
                            }
                        }
                    },
                    scales: {
                        x: { grid: { display: false }, ticks: { color: '#8c7671' } },
                        y: { 
                            ticks: { 
                                color: '#8c7671',
                                callback: function(value) { return value + ' 萬'; }
                            }
                        }
                    }
                }
            });
        }
    </script>
