    <script>
        document.addEventListener('DOMContentLoaded', function() {
            // 抓取所有輸入元件
            const inputs = {
                age: { num: document.getElementById('num-age'), range: document.getElementById('range-age'), txt: document.getElementById('val-age') },
                retire: { num: document.getElementById('num-retire'), range: document.getElementById('range-retire'), txt: document.getElementById('val-retire') },
                death: { num: document.getElementById('num-death'), range: document.getElementById('range-death'), txt: document.getElementById('val-death') },
                expense: { num: document.getElementById('num-expense'), txt: document.getElementById('val-expense') }
            };

            const resultArea = document.getElementById('result-area');

            // 同步數字輸入框與拉桿
            function setupSync(elementObj) {
                if(!elementObj.range) return;
                
                elementObj.num.addEventListener('input', function() {
                    elementObj.range.value = this.value;
                    calculateRetirement();
                });
                elementObj.range.addEventListener('input', function() {
                    elementObj.num.value = this.value;
                    calculateRetirement();
                });
            }

            setupSync(inputs.age);
            setupSync(inputs.retire);
            setupSync(inputs.death);
            inputs.expense.num.addEventListener('input', calculateRetirement);

            // 核心計算與顯示邏輯
            function calculateRetirement() {
                const age = parseInt(inputs.age.num.value) || 0;
                const retire = parseInt(inputs.retire.num.value) || 0;
                const death = parseInt(inputs.death.num.value) || 0;
                const expense = parseInt(inputs.expense.num.value) || 0;

                // 更新畫面上的數字顯示
                inputs.age.txt.innerText = age;
                inputs.retire.txt.innerText = retire;
                inputs.death.txt.innerText = death;
                inputs.expense.txt.innerText = expense.toLocaleString();

                // 邏輯防呆
                if (retire > age && death > retire && expense > 0) {
                    const yearsToRetire = retire - age;
                    const retirementYears = death - retire;
                    const totalFund = retirementYears * expense * 12;

                    // 填入詳細資訊
                    document.getElementById('res-years-to-retire').innerText = yearsToRetire;
                    document.getElementById('res-retirement-years').innerText = retirementYears;
                    document.getElementById('res-total-fund').innerText = totalFund.toLocaleString();

                    // 顯示結果區塊
                    resultArea.classList.add('active');
                } else {
                    // 若條件不符合則隱藏結果
                    resultArea.classList.remove('active');
                }
            }

            // 初始化第一次計算
            calculateRetirement();
        });
    </script>
