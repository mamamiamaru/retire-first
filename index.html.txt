        function renderNativeChart() {
            const container = document.getElementById('native-bars-container');
            container.innerHTML = ''; 

            const startAge = currentCalcData.age;
            const endAge = currentCalcData.retire;
            let startMoney = currentCalcData.saved;
            const finalFundGap = currentCalcData.finalFund;
            const years = endAge - startAge;

            let rate = 0;
            let monthlyInvest = 0;
            let barColor = 'var(--rose)';

            if (currentCalcData.isDebt) {
                rate = 0.07;
                startMoney = 0; 
                if (currentMode === 'flat') { rate = 0; monthlyInvest = currentCalcData.saveFlat; barColor = '#564743'; }
                else if (currentMode === '7') { rate = 0.07; monthlyInvest = currentCalcData.save7; barColor = 'var(--rose)'; }
                else if (currentMode === '10') { rate = 0.10; monthlyInvest = currentCalcData.save10; barColor = '#2b5c8f'; }
            } else if (finalFundGap === 0) {
                rate = 0.07;
                monthlyInvest = 0;
                barColor = '#2b5c8f';
            } else {
                if(currentMode === 'flat') { rate = 0; monthlyInvest = currentCalcData.saveFlat; barColor = '#564743'; }
                else if(currentMode === '7') { rate = 0.07; monthlyInvest = currentCalcData.save7; barColor = 'var(--rose)'; }
                else if(currentMode === '10') { rate = 0.10; monthlyInvest = currentCalcData.save10; barColor = '#2b5c8f'; }
            }

            let ageNodes = [];
            if (years <= 5) {
                for(let i = 0; i <= years; i++) ageNodes.push(startAge + i);
            } else {
                ageNodes.push(startAge);
                ageNodes.push(startAge + Math.round(years * 0.25));
                ageNodes.push(startAge + Math.round(years * 0.5));
                ageNodes.push(startAge + Math.round(years * 0.75));
                ageNodes.push(endAge);
            }

            let maxTotalBalance = startMoney;
            const monthlyRateForMax = rate / 12;
            for(let m = 0; m < (years * 12); m++) {
                maxTotalBalance = (maxTotalBalance + monthlyInvest) * (1 + monthlyRateForMax);
            }
            if(maxTotalBalance <= 0) maxTotalBalance = 1;

            ageNodes.forEach(nodeAge => {
                let currentBalance = startMoney;
                let activeMonths = (nodeAge - startAge) * 12;
                const mRate = rate / 12;

                for(let m = 0; m < activeMonths; m++) {
                    currentBalance = (currentBalance + monthlyInvest) * (1 + mRate);
                }

                let valueInWan = Math.round(currentBalance / 10000);
                let heightPct = Math.min(100, Math.max(4, (currentBalance / maxTotalBalance) * 100));

                const barWrapper = document.createElement('div');
                barWrapper.className = 'chart-bar-wrapper';

                barWrapper.innerHTML = `
                    <div class="bar-value">${valueInWan}萬</div>
                    <div class="chart-bar" style="height: 0%; background-color: ${barColor};"></div>
                    <div class="bar-label">${nodeAge}歲</div>
                `;

                container.appendChild(barWrapper);

                setTimeout(() => {
                    const barEl = barWrapper.querySelector('.chart-bar');
                    if(barEl) barEl.style.height = heightPct + '%';
                }, 50);
            });
        }
